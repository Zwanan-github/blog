    // å¹¶æŸ¥é›†
    int f[26];

    int find(int x) {
        if (f[x] != x) f[x] = find(f[x]);
        return f[x];
    }

    // åˆå¹¶ä¸”è·¯å¾„å‹ç¼©
    void merge(int a, int b) {

            f[fb] = fa;
        } else {
            f[fa] = fb;
        }
    }

    string smallestEquivalentString(string s1, string s2, string baseStr) {
        int n = s1.size();
        for (int i = 0; i < 26; ++ i) f[i] = i;

            merge((s1[i] - 'a'), (s2[i] - 'a'));
        }
        string s = "";
        for (int i = 0; i < baseStr.size(); ++ i) {
            s += 'a' + f[find(baseStr[i] - 'a')];

    }

class Solution {
public:
    string clearStars(string s) {
        int idx = 0;
        int n = s.size();
        vector<int> stack[26];
        for (int i = 0; i < n; ++ i) {
            if (s[i] != '*') {
                stack[s[i] - 'a'].push_back(i);
                continue;
            }
            for (auto& ve : stack) {
                if (!ve.empty()) {
                    s[ve.back()] = '*';
                    ve.pop_back();
                    break;
                }
            }
        }
        erase(s, '*');
        return s;

            dfs(i, n);
